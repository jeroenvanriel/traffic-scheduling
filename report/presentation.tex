\begin{textblock*}{10cm}(1cm,4cm)
\begin{itemize}
  \item Try to minimize total delay
\begin{align*}
  &\min_y \sum_i y_i - a_i \\
  &\text{ s.t. time slots are disjoint}
\end{align*}
  \vspace{0.7em}
  \item<2> Ordering decisions $\rightarrow$ Binary variables
  \item<2> Mixed-Integer Linear Program (MILP)
\end{itemize}
\end{textblock*}
\end{frame}
